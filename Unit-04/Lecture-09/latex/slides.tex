\section{Exercises}
\label{sec:exercises}


\begin{frame}{Exercise 1: Metric choice}
  \small
  Target is continuous (price). Should you use accuracy?
\end{frame}

\begin{frame}{Solution 1}
  \begin{itemize}
    \item No; accuracy is for classification.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Overfitting sign}
  \small
  Train RMSE=5, test RMSE=20. What does it suggest?
\end{frame}

\begin{frame}{Solution 2}
  \begin{itemize}
    \item Overfitting; try simpler model or regularization.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 3: Communication}
  \small
  Name one plot to communicate regression quality.
\end{frame}

\begin{frame}{Solution 3}
  \begin{itemize}
    \item Predicted vs actual scatter; residual plot.
  \end{itemize}
\end{frame}
