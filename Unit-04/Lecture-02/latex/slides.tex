\section{Model}
\label{sec:model}


\begin{frame}{Model: Key Points}
      \begin{itemize}[<+->]
        \item y = b0 + b1 x + error
\item Slope: expected change in y for 1-unit increase in x
\item Intercept: predicted y at x=0 (interpret carefully)
      \end{itemize}
    \end{frame}

\begin{frame}{Model: Key Formula}
  \[ y = \beta_0 + \beta_1 x + \epsilon \]
\end{frame}

\section{Fit and Diagnostics}
\label{sec:diag}


\begin{frame}{Fit and Diagnostics: Key Points}
      \begin{itemize}[<+->]
        \item Look at residual plots for patterns
\item Outliers can dominate the fitted line
\item High $R^2$ does not guarantee a good model
      \end{itemize}
    \end{frame}

\section{Exercises}
\label{sec:exercises}


\begin{frame}{Exercise 1: Prediction}
  \small
  Model: yhat = 10 + 2x. Predict y when x=7.
\end{frame}

\begin{frame}{Solution 1}
  \begin{itemize}
    \item yhat = 24
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Residual}
  \small
  If actual y=20 at x=7, compute residual.
\end{frame}

\begin{frame}{Solution 2}
  \begin{itemize}
    \item 20-24 = -4
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 3: Interpret slope}
  \small
  Slope is 5 thousand INR per extra room. Interpret.
\end{frame}

\begin{frame}{Solution 3}
  \begin{itemize}
    \item Each extra room increases predicted price by ~5k INR (on average).
  \end{itemize}
\end{frame}
