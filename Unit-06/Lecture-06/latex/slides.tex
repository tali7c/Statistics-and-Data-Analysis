\section{ADF Test}
\label{sec:adf}


\begin{frame}{ADF Test: Key Points}
      \begin{itemize}[<+->]
        \item H0: unit root (non-stationary)
\item H1: stationary
\item Small p-value -> reject H0
      \end{itemize}
    \end{frame}

\section{Interpretation}
\label{sec:interp}


\begin{frame}{Interpretation: Key Points}
      \begin{itemize}[<+->]
        \item If non-stationary, difference and test again
\item Seasonality can require seasonal differencing
\item Use ACF/PACF + diagnostics too
      \end{itemize}
    \end{frame}

\section{Exercises}
\label{sec:exercises}


\begin{frame}{Exercise 1: ADF null}
  \small
  What is H0 in ADF?
\end{frame}

\begin{frame}{Solution 1}
  \begin{itemize}
    \item Unit root; non-stationary.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Decision}
  \small
  If p=0.02 at alpha=0.05, what do you conclude?
\end{frame}

\begin{frame}{Solution 2}
  \begin{itemize}
    \item Reject H0; evidence of stationarity.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 3: Next step}
  \small
  If p=0.6, what next step?
\end{frame}

\begin{frame}{Solution 3}
  \begin{itemize}
    \item Difference and test again; consider seasonal differencing.
  \end{itemize}
\end{frame}
