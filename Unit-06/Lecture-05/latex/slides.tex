\section{Stationarity}
\label{sec:stat}


\begin{frame}{Stationarity: Key Points}
      \begin{itemize}[<+->]
        \item Mean/variance roughly constant
\item Autocorrelation depends on lag only
\item Trend/seasonality often implies non-stationarity
      \end{itemize}
    \end{frame}

\section{Fixes}
\label{sec:fix}


\begin{frame}{Fixes: Key Points}
      \begin{itemize}[<+->]
        \item Differencing removes trend
\item Seasonal differencing removes seasonality
\item Log transform can stabilize variance
      \end{itemize}
    \end{frame}

\section{Exercises}
\label{sec:exercises}


\begin{frame}{Exercise 1: Trend}
  \small
  Is a strong upward trend likely stationary?
\end{frame}

\begin{frame}{Solution 1}
  \begin{itemize}
    \item No; mean changes over time.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Variance change}
  \small
  If fluctuations grow over time, is variance constant?
\end{frame}

\begin{frame}{Solution 2}
  \begin{itemize}
    \item No; non-stationary variance.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 3: Fix choice}
  \small
  Name one fix for non-stationary mean.
\end{frame}

\begin{frame}{Solution 3}
  \begin{itemize}
    \item Differencing.
  \end{itemize}
\end{frame}
