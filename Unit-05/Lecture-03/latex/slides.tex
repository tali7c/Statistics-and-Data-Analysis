\begin{frame}{PCA Intuition: Key Points}
      \begin{itemize}[<+->]
        \item Find new axes (components) capturing maximum variance

\section{Exercises}
\label{sec:exercises}


\begin{frame}{Exercise 1: Scaling}
  \small
  Why scale features before PCA?
\end{frame}

\begin{frame}{Solution 1}
  \begin{itemize}
    \item To prevent large-unit features dominating variance.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Components}
  \small
  Are PCA components original features?
\end{frame}

\begin{frame}{Solution 2}
  \begin{itemize}
    \item No; they are linear combinations.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 3: Choosing k}
  \small
  If first 2 PCs explain 88\% and you need 90\%, what do you do?
\end{frame}

\begin{frame}{Solution 3}
  \begin{itemize}
    \item Add next PC(s) until target reached.
  \end{itemize}
\end{frame}
