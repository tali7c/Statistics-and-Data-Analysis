\section{Filter Methods}
\label{sec:filter}


\begin{frame}{Filter Methods: Key Points}
      \begin{itemize}[<+->]
        \item Fast scoring without training many models
\item Examples: variance threshold, correlation with target
\item May miss interactions
      \end{itemize}
    \end{frame}

\section{Wrapper/Embedded}
\label{sec:wrap}


\begin{frame}{Wrapper/Embedded: Key Points}
      \begin{itemize}[<+->]
        \item Wrapper: search subsets using a model (slow)
\item Embedded: selection during training (lasso, trees)
      \end{itemize}
    \end{frame}

\section{Exercises}
\label{sec:exercises}


\begin{frame}{Exercise 1: Low variance}
  \small
  If a feature is almost constant, keep it?
\end{frame}

\begin{frame}{Solution 1}
  \begin{itemize}
    \item Usually no; low variance adds little information.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Redundant features}
  \small
  Two features have corr=0.99. What might you do?
\end{frame}

\begin{frame}{Solution 2}
  \begin{itemize}
    \item Drop one or use regularization/PCA.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 3: Wrapper trade-off}
  \small
  Why is RFE slower than filters?
\end{frame}

\begin{frame}{Solution 3}
  \begin{itemize}
    \item It trains many models on many subsets.
  \end{itemize}
\end{frame}
