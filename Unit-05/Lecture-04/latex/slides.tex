\label{sec:lda}


\begin{frame}{LDA: Key Points}
      \begin{itemize}[<+->]

\section{Exercises}
\label{sec:exercises}


\begin{frame}{Exercise 1: Supervised?}
  \small
  Is PCA supervised? Is LDA supervised?
\end{frame}

\begin{frame}{Solution 1}
  \begin{itemize}
    \item PCA is unsupervised; LDA is supervised.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 2: Goal}
  \small
  What does PCA optimize vs LDA (intuition)?
\end{frame}

\begin{frame}{Solution 2}
  \begin{itemize}
    \item PCA: variance captured; LDA: class separability.
  \end{itemize}
\end{frame}

\begin{frame}{Exercise 3: Use case}
  \small
  Labeled A/B/C data, want 2D plot separating classes. PCA or LDA?
\end{frame}

\begin{frame}{Solution 3}
  \begin{itemize}
    \item LDA (uses labels for separation).
  \end{itemize}
\end{frame}
